import java.util.*;
 
public class CountingSort {
    public int[] countingSort(int[] A, int n) {
        // write code here、
        int[] x=new int[1000];
        for(int i=0;i<1000;i++){
            x[i]=0;
        }
        for(int i=0;i<n;i++){
            x[A[i]]++;
        }
        int k=0;
        for(int i=0;i<1000;i++){
            for(int j=0;j<x[i];j++){
                A[k++]=i;
            }
        }
        return A;
    }
}
